
void CBaseObjectBLL::GetRotateArcXY( StIXYZPoint stStart, StIXYZPoint stStop, FLOAT& fArc )
{
	double dK =(stStop.Y - stStart.Y) / (stStop.X - stStart.X);
	double dArc = std::atan(dK) * 180 / PI;
	fArc = (FLOAT)dArc;
}

void CBaseObjectBLL::GetRotateArcXZ( StIXYZPoint stStart, StIXYZPoint stStop, FLOAT& fArc )
{
	double dK =(stStop.Z - stStart.Z) / (stStop.X - stStart.X);
	double dArc = std::atan(dK) * 180 / PI;
	fArc = (FLOAT)dArc;
}

void CBaseObjectBLL::GetRotateArcYZ( StIXYZPoint stStart, StIXYZPoint stStop, FLOAT& fArc )
{
	double dK =(stStop.Y - stStart.Y) / (stStop.Z - stStart.Z);
	double dArc = std::atan(dK) * 180 / PI;
	fArc = (FLOAT)dArc;
}

void CBaseObjectBLL::GetLengthXY( StIXYZPoint stStart, StIXYZPoint stStop, double& fLength )
{
	fLength = sqrt(pow(stStop.X - stStart.X, 2) + pow(stStop.Y - stStart.Y, 2) );
}

void CBaseObjectBLL::GetLengthXZ( StIXYZPoint stStart, StIXYZPoint stStop, double& fLength )
{
	fLength = sqrt((stStop.X - stStart.X)* (stStop.X - stStart.X) + (stStop.Z - stStart.Z)* (stStop.Z - stStart.Z) );
}

void CBaseObjectBLL::GetLengthYZ( StIXYZPoint stStart, StIXYZPoint stStop, double& fLength )
{
	fLength = sqrt((stStop.Y - stStart.Y)* (stStop.Y - stStart.Y) + (stStop.Z - stStart.Z)* (stStop.Z - stStart.Z) );
}
